#define SEMGBoxFlattenAppearance(box) \
    do { \
        box.layer.shadowColor = nil; \
        box.layer.shadowOffset = CGSizeZero; \
        box.layer.shadowRadius = 0; \
        box.layer.shadowOpacity = 0; \
        box.borderStyle = MGBorderNone; \
    } while(0)
